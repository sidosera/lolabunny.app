include("${CMAKE_SOURCE_DIR}/cmake/swiftpm.cmake")

set(APP_SERVER_PACKAGE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/src")
set(APP_SERVER_ICON_SOURCE "${PROJECT_SOURCE_DIR}/bunny.png")
set(APP_SERVER_ENTITLEMENTS "${APP_SERVER_PACKAGE_DIR}/${LOLABUNNY_APP_NAME}.entitlements")
set(APP_SERVER_BUNDLE_SCRIPT "${PROJECT_SOURCE_DIR}/cmake/bundle.cmake")
set(APP_SERVER_BUNDLE_DIR "${PROJECT_SOURCE_DIR}/target/bundle")
set(APP_SERVER_APP_BUNDLE "${APP_SERVER_BUNDLE_DIR}/${LOLABUNNY_APP_NAME}.app")
set(APP_SERVER_DMG_STAGING_DIR "${CMAKE_BINARY_DIR}/dmg-staging")
set(APP_SERVER_VERSION_FILE "${PROJECT_SOURCE_DIR}/.version")

set(APP_SERVER_APP_VERSION "dev")
if(EXISTS "${APP_SERVER_VERSION_FILE}")
    file(STRINGS "${APP_SERVER_VERSION_FILE}" _app_server_version LIMIT_COUNT 1)
    if(NOT _app_server_version STREQUAL "")
        set(APP_SERVER_APP_VERSION "${_app_server_version}")
    endif()
endif()

set(APP_SERVER_DMG_NAME "${LOLABUNNY_APP_NAME}-${APP_SERVER_APP_VERSION}-${ARCH_SWIFT_ARCH}.dmg")
set(APP_SERVER_DMG_PATH "${PROJECT_SOURCE_DIR}/${APP_SERVER_DMG_NAME}")

find_program(APP_SERVER_SWIFT_EXECUTABLE NAMES swift REQUIRED)
find_program(APP_SERVER_CODESIGN_EXECUTABLE NAMES codesign REQUIRED)
find_program(APP_SERVER_STRIP_EXECUTABLE NAMES strip REQUIRED)
find_program(APP_SERVER_SIPS_EXECUTABLE NAMES sips REQUIRED)
find_program(APP_SERVER_ICONUTIL_EXECUTABLE NAMES iconutil REQUIRED)
find_program(APP_SERVER_HDIUTIL_EXECUTABLE NAMES hdiutil REQUIRED)
find_program(APP_SERVER_OPEN_EXECUTABLE NAMES open REQUIRED)

swiftpm_add_build_target(
    TARGET_NAME app.build
    SWIFT_EXECUTABLE "${APP_SERVER_SWIFT_EXECUTABLE}"
    PACKAGE_PATH "${APP_SERVER_PACKAGE_DIR}"
    SCRATCH_PATH "${SWIFT_SCRATCH_PATH}"
    PRODUCT "${LOLABUNNY_APP_NAME}"
    TRIPLE "${SWIFTPM_TARGET_TRIPLE}"
    WORKING_DIRECTORY "${PROJECT_SOURCE_DIR}"
)

add_custom_target(app.bundle
    DEPENDS app.build app-server.build
    COMMAND "${CMAKE_COMMAND}"
        -DAPP_NAME=${LOLABUNNY_APP_NAME}
        -DSERVER_BIN=${LOLABUNNY_SERVER_BIN}
        -DRUST_BIN=$<TARGET_FILE:${LOLABUNNY_SERVER_BIN}>
        -DGADGET_DIR=${APP_SERVER_PACKAGE_DIR}
        -DSWIFT_SCRATCH_PATH=${SWIFT_SCRATCH_PATH}
        -DSWIFT_TRIPLE=${SWIFTPM_TARGET_TRIPLE}
        -DBUNDLE_DIR=${APP_SERVER_BUNDLE_DIR}
        -DICON_SOURCE=${APP_SERVER_ICON_SOURCE}
        -DENTITLEMENTS=${APP_SERVER_ENTITLEMENTS}
        -DCODESIGN_IDENTITY=${CODESIGN_IDENTITY}
        -DSOURCE_DIR=${PROJECT_SOURCE_DIR}
        -DSWIFT_EXECUTABLE=${APP_SERVER_SWIFT_EXECUTABLE}
        -DCODESIGN_EXECUTABLE=${APP_SERVER_CODESIGN_EXECUTABLE}
        -DSTRIP_EXECUTABLE=${APP_SERVER_STRIP_EXECUTABLE}
        -DSIPS_EXECUTABLE=${APP_SERVER_SIPS_EXECUTABLE}
        -DICONUTIL_EXECUTABLE=${APP_SERVER_ICONUTIL_EXECUTABLE}
        -P "${APP_SERVER_BUNDLE_SCRIPT}"
    WORKING_DIRECTORY "${PROJECT_SOURCE_DIR}"
    VERBATIM
)

add_custom_target(app.dmg
    DEPENDS app.bundle
    COMMAND "${CMAKE_COMMAND}" -E rm -rf "${APP_SERVER_DMG_STAGING_DIR}"
    COMMAND "${CMAKE_COMMAND}" -E make_directory "${APP_SERVER_DMG_STAGING_DIR}"
    COMMAND "${CMAKE_COMMAND}" -E copy_directory
        "${APP_SERVER_APP_BUNDLE}"
        "${APP_SERVER_DMG_STAGING_DIR}/${LOLABUNNY_APP_NAME}.app"
    COMMAND "${CMAKE_COMMAND}" -E create_symlink
        /Applications
        "${APP_SERVER_DMG_STAGING_DIR}/Applications"
    COMMAND "${APP_SERVER_HDIUTIL_EXECUTABLE}" create
        -volname "${LOLABUNNY_APP_NAME}"
        -srcfolder "${APP_SERVER_DMG_STAGING_DIR}"
        -ov
        -format UDZO
        "${APP_SERVER_DMG_PATH}"
    COMMAND "${CMAKE_COMMAND}" -E rm -rf "${APP_SERVER_DMG_STAGING_DIR}"
    VERBATIM
)

add_custom_target(app.install
    DEPENDS app.bundle
    COMMAND "${CMAKE_COMMAND}" -E rm -rf "/Applications/${LOLABUNNY_APP_NAME}.app"
    COMMAND "${CMAKE_COMMAND}" -E copy_directory
        "${APP_SERVER_APP_BUNDLE}"
        "/Applications/${LOLABUNNY_APP_NAME}.app"
    VERBATIM
)

add_custom_target(app.run
    DEPENDS app.bundle
    COMMAND "${APP_SERVER_OPEN_EXECUTABLE}" "${APP_SERVER_APP_BUNDLE}"
    VERBATIM
)

add_custom_target(app.clean
    COMMAND "${APP_SERVER_SWIFT_EXECUTABLE}" package
        --package-path "${APP_SERVER_PACKAGE_DIR}"
        --scratch-path "${SWIFT_SCRATCH_PATH}"
        clean
    COMMAND "${CMAKE_COMMAND}" -E rm -rf
        "${APP_SERVER_BUNDLE_DIR}"
        "${APP_SERVER_DMG_STAGING_DIR}"
        "${APP_SERVER_DMG_PATH}"
        "${APP_SERVER_PACKAGE_DIR}/.build"
    WORKING_DIRECTORY "${PROJECT_SOURCE_DIR}"
    VERBATIM
)

if(BUILD_TESTING)
    swiftpm_add_test(
        TEST_NAME test.app
        SWIFT_EXECUTABLE "${APP_SERVER_SWIFT_EXECUTABLE}"
        PACKAGE_PATH "${APP_SERVER_PACKAGE_DIR}"
        SCRATCH_PATH "${SWIFT_SCRATCH_PATH}"
        WORKING_DIRECTORY "${PROJECT_SOURCE_DIR}"
        LABEL "app"
    )
endif()
